package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;

import android.os.Bundle;
import android.view.View;
import android.widget.AdapterView;
import android.widget.ArrayAdapter;
import android.widget.ListView;
import android.widget.Toast;

import java.util.ArrayList;
import java.util.List;

public class MainActivity extends AppCompatActivity {
ListView list;

List<String> arr ;



//String []food={"apple","bnana","grapes","orange"};
    @Override
    protected void onCreate(Bundle savedInstanceState) {
        arr = new ArrayList<String>();
        arr.add("banna");
        arr.add("orange");
        arr.add("peache");
        arr.add("kiwi");

        System.out.println(arr);
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        list=findViewById(R.id.aa);
        final ArrayAdapter <String> stringArrayAdapter=new ArrayAdapter<>(MainActivity.this,android.R.layout.simple_list_item_1,arr);
        list.setAdapter(stringArrayAdapter);

        list.setAdapter(stringArrayAdapter);
        list.setOnItemClickListener(new AdapterView.OnItemClickListener() {
            @Override
            public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
               // String msg=food.charAt
                Toast.makeText(MainActivity.this, arr.get(position)+ "", Toast.LENGTH_SHORT).show();
            }
        });


    }
}
